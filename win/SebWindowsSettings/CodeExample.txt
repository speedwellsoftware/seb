        private static SEBSettings DeserializeFromEnryptedXML(string decriptedSebClientSettings)
        {
            decriptedSebClientSettings = decriptedSebClientSettings.Trim();

            //TextReader textReader = new StreamReader(sebClientConfigPath);
            SEBSettings sEBSettings;
            sEBSettings = (SEBSettings)deserializer.Deserialize(memStream);

        }
